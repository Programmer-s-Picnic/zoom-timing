<script>
(function(){

const JSON_URL = "today.json";

const els = {
  timing: document.getElementById("zoomTimingValue"),
  topic: document.getElementById("classTopicValue"),
  note: document.getElementById("specialNoteValue"),
  joinBtn: document.getElementById("joinTodayBtn")
};

/* -------------------------------
   🔥 NEW: RENDER TEXT LINKS
--------------------------------*/
function renderTextLinks(items){
  const container = document.getElementById("dynamicLinksContainer");
  if(!container) return;

  if(!Array.isArray(items) || !items.length){
    container.innerHTML = "<p>No links available</p>";
    return;
  }

  container.innerHTML = items.map(item => `
    <a class="portal-link"
       href="${item.link}"
       target="_blank"
       rel="noopener noreferrer">
      <span>${item.text}</span>
      <span>Open →</span>
    </a>
  `).join("");
}

/* -------------------------------
   EXISTING LOGIC
--------------------------------*/
function applyData(data){

  const timing = data?.zoom?.timing || "Today";
  const topic = data?.class?.topic || "AI-ML Class";
  const note = data?.note?.text || "";
  const link = data?.zoom?.link || "#";
  const joinText = data?.zoom?.joinText || "Join Class";

  els.timing.textContent = timing;
  els.topic.textContent = topic;
  els.note.textContent = note;

  els.joinBtn.href = link;
  els.joinBtn.textContent = joinText;

  /* ✅ CALL HERE */
  renderTextLinks(data.textLinks);
}

/* -------------------------------
   FETCH JSON
--------------------------------*/
async function loadData(){
  try{
    const res = await fetch(JSON_URL);
    const data = await res.json();
    applyData(data);
  }catch(e){
    console.error(e);
  }
}

loadData();

})();
</script>
